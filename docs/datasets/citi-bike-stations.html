Streamlit 기본 코드

import pandas as pd
import streamlit as st

URL = "https://thinkervis.github.io/free-api-data-science-edu/data/citi_bike_station_snapshot.csv"

st.title("citi_bike_station_snapshot.csv")
df = pd.read_csv(URL)
st.write(df.shape)
st.dataframe(df.head(100))

# 숫자 컬럼이 있으면 빠르게 차트 확인
num_cols = df.select_dtypes("number").columns.tolist()
if num_cols:
    st.line_chart(df[num_cols[:3]])
